1//! [`FlattenObjects`] is a container that stores numbered objects.
2//!
3//! Objects can be added to the [`FlattenObjects`], a unique ID will be assigned
4//! to the object. The ID can be used to retrieve the object later.
5//!
6//! # Example
7//!
8//! ```
9//! use flatten_objects::FlattenObjects;
10//!
11//! let mut objects = FlattenObjects::<u32, 20>::new();
12//!
13//! // Add `23` 10 times and assign them IDs from 0 to 9.
14//! for i in 0..=9 {
15//!     objects.add_at(i, 23).unwrap();
16//!     assert!(objects.is_assigned(i));
17//! }
18//!
19//! // Remove the object with ID 6.
20//! assert_eq!(objects.remove(6), Some(23));
21//! assert!(!objects.is_assigned(6));
22//!
23//! // Add `42` (the ID 6 is available now).
24//! let id = objects.add(42).unwrap();
25//! assert_eq!(id, 6);
26//! assert!(objects.is_assigned(id));
27//! assert_eq!(objects.get(id), Some(&42));
28//! assert_eq!(objects.remove(id), Some(42));
29//! assert!(!objects.is_assigned(id));
30//! ```
31
32#![no_std]
33
34use bitmaps::{Bitmap, Bits, BitsImpl, Iter};
35use core::mem::MaybeUninit;
36
37/// A container that stores numbered objects.
38///
39/// See the [crate-level documentation](crate) for more details.
40///
41/// `CAP` is the maximum number of objects that can be held. It also equals the
42/// maximum ID that can be assigned plus one. Currently, `CAP` must not be
43/// greater than 1024.
44pub struct FlattenObjects<T, const CAP: usize>
45where
46    BitsImpl<{ CAP }>: Bits,
47{
48    objects: [MaybeUninit<T>; CAP],
49    id_bitmap: Bitmap<CAP>,
50    count: usize,
51}
52
53impl<T, const CAP: usize> FlattenObjects<T, CAP>
54where
55    BitsImpl<{ CAP }>: Bits,
56{
57    /// Creates a new empty `FlattenObjects`.
58    ///
59    /// # Panics
60    ///
61    /// Panics if `CAP` is greater than 1024.
62    ///
63    /// # Example
64    ///
65    /// ```
66    /// use flatten_objects::FlattenObjects;
67    ///
68    /// let objects = FlattenObjects::<u32, 20>::new();
69    /// assert_eq!(objects.capacity(), 20);
70    /// ```
71    ///
72    /// ```compile_fail
73    /// use flatten_objects::FlattenObjects;
74    ///
75    /// let objects = FlattenObjects::<u32, 1025>::new();
76    /// ```
77    pub const fn new() -> Self {
78        Self {
79            objects: [const { MaybeUninit::uninit() }; CAP],
80            // SAFETY: zero initialization is OK for `id_bitmap` (an array of integers).
81            id_bitmap: unsafe { MaybeUninit::zeroed().assume_init() },
82            count: 0,
83        }
84    }
85
86    /// Returns the maximum number of objects that can be held.
87    ///
88    /// It also equals the maximum ID that can be assigned plus one.
89    ///
90    /// # Example
91    ///
92    /// ```
93    /// use flatten_objects::FlattenObjects;
94    ///
95    /// let objects = FlattenObjects::<u32, 20>::new();
96    /// assert_eq!(objects.capacity(), 20);
97    /// ```
98    #[inline]
99    pub const fn capacity(&self) -> usize {
100        CAP
101    }
102
103    /// Returns the number of objects that have been added.
104    ///
105    /// # Example
106    ///
107    /// ```
108    /// use flatten_objects::FlattenObjects;
109    ///
110    /// let mut objects = FlattenObjects::<u32, 20>::new();
111    /// assert_eq!(objects.count(), 0);
112    /// objects.add(23);    // Assign ID 0.
113    /// assert_eq!(objects.count(), 1);
114    /// objects.add(42);    // Assign ID 1.
115    /// assert_eq!(objects.count(), 2);
116    /// objects.remove(0);  // ID 0 is assigned.
117    /// assert_eq!(objects.count(), 1);
118    /// objects.remove(10); // ID 10 is not assigned.
119    /// assert_eq!(objects.count(), 1);
120    /// ```
121    #[inline]
122    pub const fn count(&self) -> usize {
123        self.count
124    }
125
126    /// Checks if the given `id` is assigned.
127    ///
128    /// Returns `false` if the `id` is out of range.
129    ///
130    /// # Example
131    ///
132    /// ```
133    /// use flatten_objects::FlattenObjects;
134    ///
135    /// let mut objects = FlattenObjects::<u32, 20>::new();
136    /// objects.add(23);        // Assign ID 0.
137    /// objects.add_at(5, 42);  // Assign ID 5.
138    /// assert!(objects.is_assigned(0));
139    /// assert!(!objects.is_assigned(1));
140    /// assert!(objects.is_assigned(5));
141    /// assert!(!objects.is_assigned(10));
142    /// ```
143    #[inline]
144    pub fn is_assigned(&self, id: usize) -> bool {
145        id < CAP && self.id_bitmap.get(id)
146    }
147
148    /// Returns the reference of the element with the given `id` if it already
149    /// be assigned. Otherwise, returns `None`.
150    ///
151    /// # Example
152    ///
153    /// ```
154    /// use flatten_objects::FlattenObjects;
155    ///
156    /// let mut objects = FlattenObjects::<u32, 20>::new();
157    /// objects.add(23);        // Assign ID 0.
158    /// objects.add_at(5, 42);  // Assign ID 5.
159    /// assert_eq!(objects.get(0), Some(&23));
160    /// assert_eq!(objects.get(1), None);
161    /// assert_eq!(objects.get(5), Some(&42));
162    /// assert_eq!(objects.get(10), None);
163    /// ```
164    #[inline]
165    pub fn get(&self, id: usize) -> Option<&T> {
166        if self.is_assigned(id) {
167            // SAFETY: the object at `id` should be initialized by `add` or
168            // `add_at`.
169            unsafe { Some(self.objects[id].assume_init_ref()) }
170        } else {
171            None
172        }
173    }
174
175    /// Returns the mutable reference of the element with the given `id` if it
176    /// exists. Otherwise, returns `None`.
177    ///
178    /// # Example
179    ///
180    /// ```
181    /// use flatten_objects::FlattenObjects;
182    ///
183    /// let mut objects = FlattenObjects::<u32, 20>::new();
184    /// objects.add(23);        // Assign ID 0.
185    /// objects.add_at(5, 42);  // Assign ID 5.
186    /// *objects.get_mut(0).unwrap() = 24;
187    /// assert_eq!(objects.get_mut(1), None);
188    /// *objects.get_mut(5).unwrap() = 43;
189    /// assert_eq!(objects.get_mut(10), None);
190    /// assert_eq!(objects.get(0), Some(&24));
191    /// assert_eq!(objects.get(5), Some(&43));
192    /// ```
193    #[inline]
194    pub fn get_mut(&mut self, id: usize) -> Option<&mut T> {
195        if self.is_assigned(id) {
196            // SAFETY: the object at `id` should be initialized by `add` or
197            // `add_at`.
198            unsafe { Some(self.objects[id].assume_init_mut()) }
199        } else {
200            None
201        }
202    }
203
204    /// Add an object and assigns it the smallest available ID.
205    ///
206    /// Returns the ID if there is one available. Otherwise, returns the object
207    /// itself wrapped in `Err`.
208    ///
209    /// # Example
210    ///
211    /// ```
212    /// use flatten_objects::FlattenObjects;
213    ///
214    /// let mut objects = FlattenObjects::<u32, 3>::new();
215    /// assert_eq!(objects.add(23), Ok(0));
216    /// assert_eq!(objects.add(42), Ok(1));
217    /// assert_eq!(objects.add(23), Ok(2));
218    /// assert_eq!(objects.add(42), Err(42));
219    /// objects.remove(1);
220    /// assert_eq!(objects.add(42), Ok(1));
221    /// ```
222    pub fn add(&mut self, value: T) -> Result<usize, T> {
223        match self.id_bitmap.first_false_index() {
224            Some(id) if id < CAP => {
225                self.count += 1;
226                self.id_bitmap.set(id, true);
227                self.objects[id].write(value);
228                Ok(id)
229            }
230            _ => Err(value),
231        }
232    }
233
234    /// Add an object with the given ID.
235    ///
236    /// Returns the ID if the object is added successfully. Otherwise, returns
237    /// the object itself wrapped in `Err`.
238    ///
239    /// # Example
240    ///
241    /// ```
242    /// use flatten_objects::FlattenObjects;
243    ///
244    /// let mut objects = FlattenObjects::<u32, 20>::new();
245    /// assert_eq!(objects.add_at(5, 23), Ok(5));
246    /// assert_eq!(objects.add_at(5, 42), Err(42));
247    /// assert_eq!(objects.add_at(20, 42), Err(42));
248    /// ```
249    pub fn add_at(&mut self, id: usize, value: T) -> Result<usize, T> {
250        if id >= CAP || self.is_assigned(id) {
251            return Err(value);
252        }
253        self.count += 1;
254        self.id_bitmap.set(id, true);
255        self.objects[id].write(value);
256        Ok(id)
257    }
258
259    /// Adds an object with the given ID, replacing and returning the old object
260    /// if the ID is already assigned.
261    ///
262    /// Returns the ID if the object is added successfully. Returns `Err(Some(old))`
263    /// if the ID is already assigned. Returns `Err(None)` if the ID is out of
264    /// range.
265    ///
266    /// # Example
267    ///
268    /// ```
269    /// use flatten_objects::FlattenObjects;
270    ///
271    /// let mut objects = FlattenObjects::<u32, 20>::new();
272    /// assert_eq!(objects.add_or_replace_at(5, 23), Ok(5));
273    /// assert_eq!(objects.add_or_replace_at(5, 42), Err(Some(23)));
274    /// assert_eq!(objects.get(5), Some(&42));
275    /// assert_eq!(objects.add_or_replace_at(20, 42), Err(None));
276    /// ```
277    pub fn add_or_replace_at(&mut self, id: usize, value: T) -> Result<usize, Option<T>> {
278        if id >= CAP {
279            return Err(None);
280        }
281
282        if self.is_assigned(id) {
283            // SAFETY: the object at `id` should be initialized by `add` or
284            // `add_at`, and can not be retrieved by `get` or `get_mut` unless
285            // it be added again.
286            let old = unsafe { Some(self.objects[id].assume_init_read()) };
287            self.objects[id].write(value);
288
289            Err(old)
290        } else {
291            self.count += 1;
292            self.id_bitmap.set(id, true);
293            self.objects[id].write(value);
294
295            Ok(id)
296        }
297    }
298
299    /// Removes and returns the object with the given ID.
300    ///
301    /// After this operation, the ID is freed and can be assigned for next
302    /// object again.
303    ///
304    /// # Example
305    ///
306    /// ```
307    /// use flatten_objects::FlattenObjects;
308    ///
309    /// let mut objects = FlattenObjects::<u32, 20>::new();
310    /// let id = objects.add(23).unwrap();
311    /// assert_eq!(objects.remove(id), Some(23));
312    /// assert!(!objects.is_assigned(id));
313    /// assert_eq!(objects.remove(id), None);
314    /// ```
315    pub fn remove(&mut self, id: usize) -> Option<T> {
316        if self.is_assigned(id) {
317            self.id_bitmap.set(id, false);
318            self.count -= 1;
319            // SAFETY: the object at `id` should be initialized by `add` or
320            // `add_at`, and can not be retrieved by `get` or `get_mut` unless
321            // it be added again.
322            unsafe { Some(self.objects[id].assume_init_read()) }
323        } else {
324            None
325        }
326    }
327
328    /// Gets an iterator over the IDs of the assigned objects.
329    ///
330    /// # Example
331    ///
332    /// ```
333    /// use flatten_objects::FlattenObjects;
334    ///
335    /// let mut objects = FlattenObjects::<u32, 20>::new();
336    /// objects.add(23);        // Assign ID 0.
337    /// objects.add_at(5, 42);  // Assign ID 5.
338    /// let mut iter = objects.ids();
339    /// assert_eq!(iter.next(), Some(0));
340    /// assert_eq!(iter.next(), Some(5));
341    /// assert_eq!(iter.next(), None);
342    /// ```
343    #[inline]
344    pub fn ids(&self) -> Iter<'_, CAP> {
345        self.id_bitmap.into_iter()
346    }
347}

358impl<T: Clone, const CAP: usize> Clone for FlattenObjects<T, CAP>
359where
360    BitsImpl<{ CAP }>: Bits,
361{
362    fn clone(&self) -> Self {
363        let mut cloned = Self::new();
364        cloned.count = self.count;
365        cloned.id_bitmap = self.id_bitmap;
366        for id in &self.id_bitmap {
367            cloned.objects[id].write(unsafe { self.objects[id].assume_init_ref() }.clone());
368        }
369        cloned
370    }
371}

373impl<T, const CAP: usize> Drop for FlattenObjects<T, CAP>
374where
375    BitsImpl<{ CAP }>: Bits,
376{
377    fn drop(&mut self) {
378        for id in &self.id_bitmap {
379            unsafe { self.objects[id].assume_init_drop() };
380        }
381    }
